1function [ytrend,ycycle] = one_sided_hp_filter_kalman(y,lambda)
 2
 3% One-sided HP filter using the Kalman filter to optimally one-sidedly 
 4% filter the series that renders the standard two-sided HP filter optimal.
 5
 6%% Inputs  
 7% y         Txn matrix, with T time series obs. and n variables
 8% lambda    scalar, smoothing parameter
 9% x_user    2xn matrix with initial values of the state estimate for each 
10%           variable in y. The underlying state vector is 2x1m hence two values are needed
11%           for each variable in y. Optional: if not entered, default backwards extrapolations based on the
12%           first two observations will be used.
13% P_user    a structural array with n elements, each a two
14%           2x2 matrix of intial MSE estimates for each
15%           variable in y.  Optional: if not entered,
16%           default matrix with large variances used.
17% discard   scalar. The first discard periods will be
18%           discarded resulting in output matrices of size
19%           (T-discard)xn. Optional: if not entered, a default
20%           value of 0 will be used.
21
22%% Outputs
23% ytrend   (T-discard)xn matrix of extracted trends for each of the n variables.
24% ycycle   (T-discard)xn matrix of deviations from the extracted trends for
25%          each of the n variables. Optional.
26
27%if nargin < 2 || isempty(lambda),  lambda = 1600; end
28[T,n] = size(y);
29ytrend = NaN(T,n);
30
31% The notation follows Chapter 13 of Hamilton, J.D. (1994). Time Series Analysis. with the exception of H, which is equivalent to his H'.
32q = 1/lambda; % signal-to-noise ratio: i.e. var eta_t / var epsilon_t
33F = [2,-1;1,0]; % state transition matrix
34H = [1,0]; % observation matrix
35Q = [q,0;0,0]; % variance-covariance matrix of the errors in the state equation
36R = 1; % variance of the error in the observation equation
37
38x_user = [];
39P_user = [];
40
41for k=1:n % run Kalman filter for each variable
42    if nargin < 4 || isempty(x_user), x=[2*y(1,k)-y(2,k); 3*y(1,k)-2*y(2,k)]; else x=x_user(:,k);end % If the user didn't provide an intial value for state estimate, extrapolate back two periods from the observations
43    if nargin < 4 || isempty(P_user), P= [1e5 0;0 1e5]; else P=P_user{k}; end % If the user didn't provide an intial value for the MSE, set a rather high one
44
45    for j=1:T
46        [x,P] = kalman_update(F,H,Q,R,y(j,k),x,P);
47        ytrend(j,k) = x(2);  % second element of the state is the estimate of the trend
48    end
49end
50
51% Compute gap in case it was requested
52if nargout==2 
53    ycycle = y-ytrend;
54end
55
56% If the user provided discard parameter
57%if nargin==5 
58%    ytrend=ytrend(discard+1:end,:);% remove the first "discard" periods from the trend series
59%    if nargout==2 % should the user have requested the gap
60%        ycycle=ycycle(discard+1:end,:);
61%    end
62%end
63
64end
